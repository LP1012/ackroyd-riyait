\begin{frame}
      \frametitle{Conclusions}
      % conclusions
      \begin{itemize}
            \item $S_N$ methods offer an efficient way to solve the transport
                  equation without involving large matrices, as opposed to $P_N$
                  methods
            \item $S_N$ methods suffer from "ray effects", which arise in
                  problems with low scattering and persist at higher discretizations
            \item Our next implementation will explore $S_N$ methods in finite
                  elements, as opposed to diamond-differencing (finite difference)
      \end{itemize}
\end{frame}
