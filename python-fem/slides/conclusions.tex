\begin{frame}
      \frametitle{Conclusions}
      % conclusions
      \begin{itemize}
            \item $S_N$ methods offer an efficient and intuitive way to solve the Boltzmann transport
                  equation.
            \item $S_N$ methods suffer from ``ray effects'', which arise in problems with low scattering and persist at higher angular discretizations.
            \item Our next implementation will explore the solution of the same set of problems using the finite element method.
      \end{itemize}
\end{frame}
